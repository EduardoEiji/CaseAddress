

CREATE TABLE People (
    Id UNIQUEIDENTIFIER NOT NULL PRIMARY KEY,
    Name NVARCHAR(255) NOT NULL,
    Document NVARCHAR(50) NOT NULL,
    PType INT NOT NULL,                
    BirthDate DATETIME2 NULL,         
    CompanyName NVARCHAR(255) NULL     
);

CREATE TABLE Addresses (
    Id UNIQUEIDENTIFIER NOT NULL PRIMARY KEY,
    Cep NVARCHAR(20) NOT NULL,
    Logradouro NVARCHAR(255) NOT NULL,
    Complemento NVARCHAR(255) NULL,
    Unidade NVARCHAR(50) NULL,
    Bairro NVARCHAR(100) NOT NULL,
    Localidade NVARCHAR(100) NOT NULL,
    Uf NVARCHAR(2) NOT NULL,
    Estado NVARCHAR(100) NOT NULL,
    Regiao NVARCHAR(100) NOT NULL,
    Ibge NVARCHAR(20) NULL,
    Gia NVARCHAR(20) NULL,
    Ddd NVARCHAR(5) NULL,
    Siafi NVARCHAR(20) NULL,
    PeopleId UNIQUEIDENTIFIER NOT NULL UNIQUE, 
    CONSTRAINT FK_Addresses_People FOREIGN KEY (PeopleId) REFERENCES People(Id) ON DELETE CASCADE
);
